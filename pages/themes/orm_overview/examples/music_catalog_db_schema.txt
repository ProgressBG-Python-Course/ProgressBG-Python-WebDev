---------------------------------------------------------
Table ARTIST
---------------------------------------------------------
id        (primary key)
name
notes


---------------------------------------------------------
Table ALBUM
---------------------------------------------------------
id          (primary key)
artist_id   (foreign key to artist table)
genre_id    (foreign key to genre table)
name
release_date
picture_url
notes


---------------------------------------------------------
Table TRACK
---------------------------------------------------------
id            (primary key)
album_id      (foreign key to albums table)
name
track_artist  (if null the track_artist = album.artist)
playtime
lyrics

---------------------------------------------------------
Table GENRES
---------------------------------------------------------
id      (primary key)
name
notes
